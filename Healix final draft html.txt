function vcResetResults() {
  ['vcDualPanel','vcClaims','vcDiff','vcAlt','vcVerdict'].forEach(id => {
    const el = document.getElementById(id);
    if (el) { el.classList.remove('show'); if(id==='vcDualPanel') el.style.display='none'; }
  });
  document.getElementById('vcProgress').style.display = 'none';
  document.getElementById('vcStatusMsg').style.display = 'none';
}

function vcRunCheck() {
  if (vcChecking) return;
  const url = document.getElementById('vcUrl').value.trim();
  if (!url) { showToast('Please paste a video URL or select a sample topic'); return; }
  if (!vcCurrentDemo) {
    const keys = Object.keys(VC_DEMOS);
    vcCurrentDemo = VC_DEMOS[keys[Math.floor(Math.random() * keys.length)]];
  }
  vcChecking = true;
  const btn = document.getElementById('vcCheckBtn');
  btn.disabled = true;
  btn.innerHTML = '<span class="vc-spinner"></span>Analysing…';
  vcResetResults();

  const prog = document.getElementById('vcProgress');
  const fill = document.getElementById('vcProgressFill');
  const msg = document.getElementById('vcStatusMsg');
  prog.style.display = 'block';
  msg.style.display = 'block';
  fill.style.width = '0%';

  const steps = [
    [10, 400, 'Fetching video metadata…'],
    [30, 1000, 'AI scanning all claims in video…'],
    [52, 1900, 'Sending to ' + vcCurrentDemo.docName + ' for review…'],
    [74, 2800, 'Cross-checking medical databases…'],
    [90, 3400, 'Compiling authenticity report…'],
    [100, 3900, 'Done!'],
  ];
  steps.forEach(([pct, delay, text]) => {
    setTimeout(() => {
      fill.style.width = pct + '%';
      msg.textContent = text;
    }, delay);
  });

  setTimeout(() => {
    vcChecking = false;
    btn.disabled = false;
    btn.innerHTML = 'Check another';
    prog.style.display = 'none';
    msg.style.display = 'none';
    vcShowResults();
    const demo = vcCurrentDemo;
    vcCurrentDemo = null;
  }, 4200);
}

    </div>`;
  document.getElementById('vcDiff').classList.add('show');

  // Alternatives
  document.getElementById('vcAltList').innerHTML = demo.alts.map((a, i) =>
    `<div class="vc-alt-item"><div class="vc-alt-num">${i+1}</div><div>${a}</div></div>`
  ).join('');
  document.getElementById('vcAlt').classList.add('show');

  // Verdict
  const vtop = document.getElementById('vcVerdictTop');
  vtop.className = 'vc-verdict-top vt-' + demo.verdictType;
  const icons = {danger:'⚡', warning:'⚠️', safe:'✅'};
  document.getElementById('vcVerdictIcon').textContent = icons[demo.verdictType] || '⚡';

  }, 300);

function handleSearch(val) {
  const results = document.getElementById('searchResults');
  if (!val.trim()) { results.style.display = 'none'; return; }
  const v = val.toLowerCase();
  const matchDocs = DOCTORS.filter(d => d.name.toLowerCase().includes(v) || d.spec.toLowerCase().includes(v));
  const matchReels = REELS.filter(r => r.title.toLowerCase().includes(v));
  let html = '';
  if (matchDocs.length) {
    html += '<div style="font-size:11px;font-weight:600;color:var(--blue-600);letter-spacing:.06em;text-transform:uppercase;margin-bottom:8px">Doctors</div>';
    html += matchDocs.map(d => renderDoctorCard(d)).join('');
  }
  if (matchReels.length) {
    html += '<div style="font-size:11px;font-weight:600;color:var(--blue-600);letter-spacing:.06em;text-transform:uppercase;margin:12px 0 8px">Health reels</div>';
    html += matchReels.map(r => renderReelCard(r)).join('');
  }
  if (!html) html = '<div class="empty-state"><p>No results for "' + val + '"</p></div>';
  results.innerHTML = html;
  results.style.display = 'block';
}

function renderChatHistory() {
  const msgs = document.getElementById('chatMessages');
  msgs.innerHTML = CHAT_HISTORY.map(m => {
    if (m.role === 'insight') return `<div class="insight-banner"><div class="insight-title">Why this matters</div><div class="insight-text">${m.text}</div></div>`;
    return `<div class="bubble-wrap ${m.role}"><div class="bubble ${m.role}">${m.text}</div><div class="bubble-time">${m.time}</div></div>`;

function sendChatMessage() {
  const input = document.getElementById('chatInput');
  const text = input.value.trim();
  if (!text) return;
  if (freeTextsLeft <= 0) { showToast('Upgrade to ₹5/mo for more texts'); return; }
  freeTextsLeft--;
  document.getElementById('freeCounter').textContent = freeTextsLeft + ' free text' + (freeTextsLeft !== 1 ? 's' : '') + ' left';
  document.getElementById('statTexts').textContent = (3 - freeTextsLeft);
  CHAT_HISTORY.push({role:'me', text, time: new Date().toLocaleTimeString([], {hour:'2-digit',minute:'2-digit'})});
  input.value = '';
  renderChatHistory();
  setTimeout(() => {
    const replies = [
      "Thank you for sharing that. Can you tell me more about when the symptoms started?",
      "Based on your history, I'd suggest scheduling a follow-up. Have you taken your medication today?",
      "That sounds manageable. Ensure you stay hydrated and rest well. Monitor for 24 hours.",
      "Please book an appointment so I can examine you properly and review your recent reports.",
    ];
    const reply = replies[Math.floor(Math.random() * replies.length)];
    CHAT_HISTORY.push({role:'them', text: reply, time: new Date().toLocaleTimeString([], {hour:'2-digit',minute:'2-digit'})});
    renderChatHistory();
  }, 1200);
}

function handleChatKey(e) {
  if (e.key === 'Enter' && !e.shiftKey) { e.preventDefault(); sendChatMessage(); }
}
