function handleSearch(val){
  const results=document.getElementById('searchResults');
  if(!val.trim()){results.style.display='none';return;}
  const v=val.toLowerCase();
  const matchDocs=DOCTORS.filter(d=>d.name.toLowerCase().includes(v)||d.spec.toLowerCase().includes(v));
  const matchReels=REELS.filter(r=>r.title.toLowerCase().includes(v));
  let html='';
  if(matchDocs.length){html+='<div style="font-size:11px;font-weight:600;color:var(--blue-600);letter-spacing:.06em;text-transform:uppercase;margin-bottom:8px">Doctors</div>';html+=matchDocs.map(d=>renderDoctorCard(d)).join('');}
  if(matchReels.length){html+='<div style="font-size:11px;font-weight:600;color:var(--blue-600);margin:12px 0 8px">Health reels</div>';html+=matchReels.map(r=>renderReelCard(r)).join('');}
  if(!html) html='<div class="empty-state"><p>No results for "'+val+'"</p></div>';
  results.innerHTML=html;results.style.display='block';
}

function renderChatHistory(){
  const msgs=document.getElementById('chatMessages');
  msgs.innerHTML=CHAT_HISTORY.map(m=>{
    if(m.role==='insight') return `<div class="insight-banner"><div class="insight-title">Why this matters</div><div class="insight-text">${m.text}</div></div>`;
    return `<div class="bubble-wrap ${m.role}"><div class="bubble ${m.role}">${m.text}</div><div class="bubble-time">${m.time}</div></div>`;

function sendChatMessage(){
  const input=document.getElementById('chatInput');
  const text=input.value.trim();
  if(!text) return;
  if(freeTextsLeft<=0){showToast('Upgrade to Standard for unlimited Q&As');return;}
  freeTextsLeft--;
  document.getElementById('freeCounter').textContent=freeTextsLeft+' free text'+(freeTextsLeft!==1?'s':'')+' left';
  document.getElementById('statTexts').textContent=(3-freeTextsLeft);
  CHAT_HISTORY.push({role:'me',text,time:new Date().toLocaleTimeString([],{hour:'2-digit',minute:'2-digit'})});
  input.value='';
  renderChatHistory();
  setTimeout(()=>{
    const replies=["Thank you for sharing. Can you tell me more about when the symptoms started?","Based on your history, I'd suggest a follow-up. Have you taken your medication today?","That sounds manageable. Stay hydrated and rest. Monitor for 24 hours.","Please book an appointment so I can examine you properly."];
    CHAT_HISTORY.push({role:'them',text:replies[Math.floor(Math.random()*replies.length)],time:new Date().toLocaleTimeString([],{hour:'2-digit',minute:'2-digit'})});
    renderChatHistory();
  },1200);
}
